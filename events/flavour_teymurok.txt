		name = flavor_teymurok.4.a

		custom_tooltip = teymurok_coalition_war_tt
		hidden_effect = {
			every_rival_country = { 
				form_coalition_against = ROOT
			}

			every_enemy_country = { 
				form_coalition_against = ROOT
			}

			random_rival_country = {
				if = {
					limit = {
						has_opinion_modifier = {
							who = ROOT
							modifier = scornfully_insulted
						}
					}
					form_coalition_against = ROOT
					declare_war_with_cb = {
						who = ROOT
						casus_belli = cb_super_badboy
					}
				}
			}
		}

		set_country_flag = teymurok_coalition_war
	}
}
